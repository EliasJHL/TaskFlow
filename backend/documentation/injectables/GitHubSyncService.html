import { Injectable, BadRequestException } from '@nestjs/common';
import { PrismaService } from '../prisma/prisma.service';

// Color palette for auto-generated labels/lists
const COLORS = ['#ef4444', '#f97316', '#eab308', '#22c55e', '#14b8a6', '#3b82f6', '#8b5cf6', '#ec4899'];

@Injectable()
export class GitHubSyncService {
    constructor(private prisma: PrismaService) {}

    private async githubGraphQL(query: string, variables: any, token: string) {
        if (!token) throw new BadRequestException('Missing GitHub access token');

        const res = await fetch('https://api.github.com/graphql', {
            method: 'POST',
            headers: {
                Authorization: `Bearer ${token}`,
                Accept: 'application/vnd.github+json',
                'Content-Type': 'application/json',
            },
            body: JSON.stringify({ query, variables }),
        });

        const data = await res.json();
        if (data.errors) throw new Error(JSON.stringify(data.errors));
        return data.data;
    }

    async listProjects(owner: string, isOrg: boolean, token: string) {
        const query = isOrg
            ? `query ($login: String!) { organization(login: $login) { projectsV2(first:50) { nodes { id number title closed url } } } }`
            : `query ($login: String!) { user(login: $login) { projectsV2(first:50) { nodes { id number title closed url } } } }`;

        const data = await this.githubGraphQL(query, { login: owner }, token);
        const nodes = isOrg ? data.organization?.projectsV2?.nodes : data.user?.projectsV2?.nodes;
        return (nodes || []).map((n: any) => ({
            id: n.id,
            number: n.number,
            title: n.title,
            url: n.url,
            closed: n.closed,
            owner,
            ownerType: isOrg ? 'organization' : 'user',
        }));
    }

    async getProjectFields(owner: string, projectNumber: number, isOrg: boolean, token: string) {
        const query = isOrg
            ? `query ($login: String!, $number: Int!) {
                organization(login: $login) {
                  projectV2(number: $number) {
                    fields(first: 50) {
                      nodes {
                        ... on ProjectV2Field { id name dataType }
                        ... on ProjectV2SingleSelectField { id name dataType options { id name color } }
                        ... on ProjectV2IterationField { id name dataType }
                      }
                    }
                  }
                }
              }`
            : `query ($login: String!, $number: Int!) {
                user(login: $login) {
                  projectV2(number: $number) {
                    fields(first: 50) {
                      nodes {
                        ... on ProjectV2Field { id name dataType }
                        ... on ProjectV2SingleSelectField { id name dataType options { id name color } }
                        ... on ProjectV2IterationField { id name dataType }
                      }
                    }
                  }
                }
              }`;

        const data = await this.githubGraphQL(query, { login: owner, number: projectNumber }, token);
        const nodes = isOrg ? data.organization?.projectV2?.fields?.nodes : data.user?.projectV2?.fields?.nodes;
        return (nodes || []).map((n: any) => ({
            id: n.id,
            name: n.name,
            dataType: n.dataType ?? 'TEXT',
            options: n.options ?? [],
        }));
    }

    async getProjectItemsWithFields(owner: string, projectNumber: number, isOrg: boolean, token: string, first = 100) {
        // Extended query: get items with field values, labels, assignees
        const query = isOrg
            ? `query ($login: String!, $number: Int!, $first: Int!) {
                organization(login: $login) {
                  projectV2(number: $number) {
                    items(first: $first) {
                      nodes {
                        id
                        fieldValues(first: 20) {
                          nodes {
                            ... on ProjectV2ItemFieldTextValue { text field { ... on ProjectV2Field { name } } }
                            ... on ProjectV2ItemFieldNumberValue { number field { ... on ProjectV2Field { name } } }
                            ... on ProjectV2ItemFieldDateValue { date field { ... on ProjectV2Field { name } } }
                            ... on ProjectV2ItemFieldSingleSelectValue { name field { ... on ProjectV2SingleSelectField { name } } }
                            ... on ProjectV2ItemFieldIterationValue { title field { ... on ProjectV2IterationField { name } } }
                          }
                        }
                        content {
                          __typename
                          ... on Issue {
                            id title body url state
                            labels(first: 20) { nodes { name color } }
                            assignees(first: 10) { nodes { login } }
                          }
                          ... on PullRequest {
                            id title body url state
                            labels(first: 20) { nodes { name color } }
                            assignees(first: 10) { nodes { login } }
                          }
                          ... on DraftIssue { id title body }
                        }
                      }
                    }
                  }
                }
              }`
            : `query ($login: String!, $number: Int!, $first: Int!) {
                user(login: $login) {
                  projectV2(number: $number) {
                    items(first: $first) {
                      nodes {
                        id
                        fieldValues(first: 20) {
                          nodes {
                            ... on ProjectV2ItemFieldTextValue { text field { ... on ProjectV2Field { name } } }
                            ... on ProjectV2ItemFieldNumberValue { number field { ... on ProjectV2Field { name } } }
                            ... on ProjectV2ItemFieldDateValue { date field { ... on ProjectV2Field { name } } }
                            ... on ProjectV2ItemFieldSingleSelectValue { name field { ... on ProjectV2SingleSelectField { name } } }
                            ... on ProjectV2ItemFieldIterationValue { title field { ... on ProjectV2IterationField { name } } }
                          }
                        }
                        content {
                          __typename
                          ... on Issue {
                            id title body url state
                            labels(first: 20) { nodes { name color } }
                            assignees(first: 10) { nodes { login } }
                          }
                          ... on PullRequest {
                            id title body url state
                            labels(first: 20) { nodes { name color } }
                            assignees(first: 10) { nodes { login } }
                          }
                          ... on DraftIssue { id title body }
                        }
                      }
                    }
                  }
                }
              }`;

        const data = await this.githubGraphQL(query, { login: owner, number: projectNumber, first }, token);
        const nodes = isOrg ? data.organization?.projectV2?.items?.nodes : data.user?.projectV2?.items?.nodes;

        return (nodes || []).map((n: any) => {
            const fieldValues: Record<string, string> = {};
            for (const fv of n.fieldValues?.nodes || []) {
                const fieldName = fv.field?.name;
                if (!fieldName) continue;
                fieldValues[fieldName] = fv.text ?? fv.number?.toString() ?? fv.date ?? fv.name ?? fv.title ?? '';
            }

            return {
                id: n.id,
                title: n.content?.title ?? 'Untitled',
                body: n.content?.body ?? null,
                type: n.content?.__typename ?? 'ITEM',
                url: n.content?.url ?? null,
                state: n.content?.state ?? null,
                labels: (n.content?.labels?.nodes || []).map((l: any) => ({ name: l.name, color: l.color })),
                assignees: (n.content?.assignees?.nodes || []).map((a: any) => a.login),
                fieldValues,
            };
        });
    }

    // backwards compatible simple version
    async getProjectItems(owner: string, projectNumber: number, isOrg: boolean, token: string, first = 50) {
        return this.getProjectItemsWithFields(owner, projectNumber, isOrg, token, first);
    }

    async connectProject(boardId: string, owner: string, projectNumber: number, isOrg: boolean, token: string) {
        const projects = await this.listProjects(owner, isOrg, token);
        const proj = projects.find((p: any) => p.number === projectNumber);
        if (!proj) throw new BadRequestException('Project not found');

        // Check if already connected
        const existing = await this.prisma.gitHubProjectSync.findUnique({ where: { board_id: boardId } });
        if (existing) {
            return this.prisma.gitHubProjectSync.update({
                where: { sync_id: existing.sync_id },
                data: { access_token: token, github_project_id: proj.id, github_project_number: proj.number, github_owner: owner },
            });
        }

        const created = await this.prisma.gitHubProjectSync.create({
            data: {
                board_id: boardId,
                github_project_id: proj.id,
                github_project_number: proj.number,
                github_owner: owner,
                github_repo: isOrg ? null : undefined,
                github_org: isOrg ? owner : null,
                access_token: token,
            },
        });

        return created;
    }

    async getSyncConfig(boardId: string) {
        return this.prisma.gitHubProjectSync.findUnique({
            where: { board_id: boardId },
            include: { field_mappings: true },
        });
    }

    async disconnectProject(boardId: string) {
        await this.prisma.gitHubProjectSync.delete({ where: { board_id: boardId } });
        return { __typename: 'Success', successMessage: 'Disconnected' };
    }

    async setupAutoMapping(boardId: string, token: string) {
        const sync = await this.prisma.gitHubProjectSync.findUniqueOrThrow({ where: { board_id: boardId } });
        const isOrg = !!sync.github_org;

        // 1. Fetch project fields
        const fields = await this.getProjectFields(sync.github_owner, sync.github_project_number, isOrg, token);

        // 2. Find Status field (single select)  common in GitHub Projects
        const statusField = fields.find((f: any) => f.name === 'Status' && f.dataType === 'SINGLE_SELECT');

        if (statusField && statusField.options?.length) {
            // Create lists for each status option if they don't exist
            const existingLists = await this.prisma.list.findMany({ where: { board_id: boardId } });
            const existingTitles = new Set(existingLists.map((l) => l.title.toLowerCase()));

            let pos = existingLists.length;
            const statusToList: Record<string, string> = {};

            for (const opt of statusField.options) {
                if (!existingTitles.has(opt.name.toLowerCase())) {
                    pos++;
                    const color = opt.color ? `#${opt.color}` : COLORS[pos % COLORS.length];
                    const newList = await this.prisma.list.create({ data: { title: opt.name, board_id: boardId, position: pos, color } });
                    statusToList[opt.name] = newList.list_id;
                } else {
                    const match = existingLists.find((l) => l.title.toLowerCase() === opt.name.toLowerCase());
                    if (match) statusToList[opt.name] = match.list_id;
                }
            }

            // Save field mapping
            await this.prisma.gitHubFieldMapping.upsert({
                where: { sync_id_github_field_id: { sync_id: sync.sync_id, github_field_id: statusField.id } },
                update: { mapping_values: statusToList, taskflow_field: 'list' },
                create: { sync_id: sync.sync_id, github_field_id: statusField.id, github_field_name: 'Status', taskflow_field: 'list', mapping_values: statusToList },
            });
        }

        return { success: true, message: 'Auto mapping configured' };
    }

    async importFromGitHub(boardId: string, token: string) {
        const sync = await this.prisma.gitHubProjectSync.findUniqueOrThrow({
            where: { board_id: boardId },
            include: { field_mappings: true },
        });
        const owner = sync.github_owner;
        const number = sync.github_project_number;
        const isOrg = !!sync.github_org;

        // First run auto-mapping to ensure lists exist
        await this.setupAutoMapping(boardId, token);

        const items = await this.getProjectItemsWithFields(owner, number, isOrg, token);

        // Reload mappings after auto setup
        const mappings = await this.prisma.gitHubFieldMapping.findMany({ where: { sync_id: sync.sync_id } });
        const statusMapping = mappings.find((m) => m.taskflow_field === 'list');
        const statusToListId: Record<string, string> = (statusMapping?.mapping_values as any) ?? {};

        // Default list if no status mapping
        let defaultList = await this.prisma.list.findFirst({ where: { board_id: boardId }, orderBy: { position: 'asc' } });
        if (!defaultList) {
            defaultList = await this.prisma.list.create({ data: { title: 'To Do', board_id: boardId, position: 1, color: '#6b7280' } });
        }

        // Fetch existing labels on the board
        const existingLabels = await this.prisma.label.findMany({ where: { board_id: boardId } });
        const labelNameToId: Record<string, string> = {};
        for (const l of existingLabels) labelNameToId[l.name.toLowerCase()] = l.label_id;

        // Fetch workspace members for assignee mapping
        const board = await this.prisma.board.findUniqueOrThrow({ where: { board_id: boardId }, select: { workspace_id: true } });
        const members = await this.prisma.workspaceMembers.findMany({ where: { workspace_id: board.workspace_id }, include: { user: true } });
        const usernameToUserId: Record<string, string> = {};
        for (const m of members) {
            if (m.user.github_username) usernameToUserId[m.user.github_username.toLowerCase()] = m.user.user_id;
            usernameToUserId[m.user.username.toLowerCase()] = m.user.user_id;
        }

        let count = 0;
        const errors: string[] = [];

        for (const it of items) {
            try {
                // Skip already synced items
                const existingSynced = await this.prisma.gitHubSyncedItem.findFirst({ where: { sync_id: sync.sync_id, github_item_id: it.id } });
                if (existingSynced) continue;

                // Determine target list from Status field
                const statusValue = it.fieldValues?.['Status'];
                const targetListId = (statusValue && statusToListId[statusValue]) ? statusToListId[statusValue] : defaultList!.list_id;

                // Position
                const lastCard = await this.prisma.card.findFirst({ where: { list_id: targetListId }, orderBy: { position: 'desc' } });
                const cardPos = lastCard ? lastCard.position + 1 : 1;

                // Due date from field if exists
                const dueDateRaw = it.fieldValues?.['Due date'] ?? it.fieldValues?.['Date'] ?? null;
                const dueDate = dueDateRaw ? new Date(dueDateRaw) : null;

                // Create card
                const card = await this.prisma.card.create({
                    data: {
                        title: it.title,
                        description: it.body ?? undefined,
                        position: cardPos,
                        list_id: targetListId,
                        due_date: dueDate && !isNaN(dueDate.getTime()) ? dueDate : undefined,
                    },
                });

                // Sync record
                await this.prisma.gitHubSyncedItem.create({ data: { sync_id: sync.sync_id, github_item_id: it.id, github_content_id: it.id, taskflow_card_id: card.card_id } });

                // Map labels
                for (const ghLabel of it.labels || []) {
                    let labelId = labelNameToId[ghLabel.name.toLowerCase()];
                    if (!labelId) {
                        const color = ghLabel.color ? `#${ghLabel.color}` : COLORS[Object.keys(labelNameToId).length % COLORS.length];
                        const newLabel = await this.prisma.label.create({ data: { name: ghLabel.name, color, board_id: boardId } });
                        labelId = newLabel.label_id;
                        labelNameToId[ghLabel.name.toLowerCase()] = labelId;
                    }
                    await this.prisma.cardLabel.create({ data: { card_id: card.card_id, label_id: labelId } }).catch(() => {});
                }

                // Map assignees
                for (const login of it.assignees || []) {
                    const userId = usernameToUserId[login.toLowerCase()];
                    if (userId) {
                        await this.prisma.cardMember.create({ data: { card_id: card.card_id, user_id: userId } }).catch(() => {});
                    }
                }

                count++;
            } catch (e: any) {
                errors.push(e.message ?? String(e));
            }
        }

        await this.prisma.gitHubProjectSync.update({ where: { sync_id: sync.sync_id }, data: { last_sync_at: new Date(), sync_status: 'SUCCESS' } });

        return { success: errors.length === 0, message: `Imported ${count} items`, itemsSynced: count, errors };
    }
}
